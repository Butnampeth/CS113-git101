#include <stdio.h> ต้องมีคำสั่งนี้ก่อนเริ่มเขียนโปรแกรม
#define ชื่อตัวแปร ค่าของตัวแปร;
**ทุกประโยคคำสั่งต้องมี ; ปิดท้าย
int main() //int จะต้องบอกว่าทำงานสำเร็จหรือไม่สำเร็จด้วยการรีเทิร์นค่า
void main() //void จะไม่บอกว่าทำงานสำเร็จหรือไม่
{
    int (4 byte) เลขจำนวนเต็ม ใช้ %d
    unsigned จะรับแต่ตัวเลขไม่รับเครื่องหมาย
    long (8 byte) เก็บค่าได้มากกว่า int
    long long เก็บค่าได้มากกว่า long

    float (4 byte) เลขจำนวนจริง ใช้ %f
    double (8 byte) เก็บค่าได้มากกว่า float ใช้ %lf

    char x = 'x';  ใช้ %c
    char name[] = "name";  ใช้ %s
    [] เรียกว่า Array รับค่าอินฟินิตี้
}

"\n" คือขึ้นบรรทัดใหม่
เวลาหารจะต้องมี .0 เพราะไม่งั้นค่าจะไม่คิดเศษ
    1.0/2 = 0.500000
    1/2 = 0.000000
1%2 หารเอาเศษ
